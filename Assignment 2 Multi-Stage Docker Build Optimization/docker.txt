# -------- Stage 1: Build Stage --------
FROM node:18 AS build

WORKDIR /usr/src/app
COPY package*.json ./
RUN npm install

COPY . .

# -------- Stage 2: Runtime Stage --------
FROM node:18-alpine

WORKDIR /usr/src/app

# Copy only the built app and node_modules from the build stage
COPY --from=build /usr/src/app /usr/src/app

EXPOSE 3000
CMD ["npm", "start"]
